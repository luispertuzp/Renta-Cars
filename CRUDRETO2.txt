Create table CLIENT (
    ID NUMBER (10) NOT NULL,
	NAME VARCHAR2(4000) NULL
    EMAIL VARCHAR(20) NULL,
    AGE NUMBER (10) NULL,
	PRIMARY KEY (ID)

);

PETICION  GET:

SELECT * FROM CLIENT


PETICION  POST:

BEGIN
    INSERT INTO CLIENT (ID,NAME,EMAIL,AGE)
    VALUES (:id,:name,:email,:age);
    :status_code:=201;
END;


PETICION  PUT:

BEGIN
UPDATE CLIENT SET NAME=:name,EMAIL=:email,AGE=:age WHERE ID=:id;
:status_code:=201;
END;


PETICION  DELETE: 
	
BEGIN
    DELETE FROM CLIENT WHERE ID=:id;
    :status_code:=204;    
END;


client/:id
SELECT * FROM CLIENT WHERE id=:id
